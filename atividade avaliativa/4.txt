#include <stdio.h>

int main() {
    double A, B, C;
    double maior, menor, meio;

    printf("Digite o valor de A: ");
    scanf("%lf", &A);

    printf("Digite o valor de B: ");
    scanf("%lf", &B);

    printf("Digite o valor de C: ");
    scanf("%lf", &C);

  
    if (A + B > C && B + C > A && C + A > B) {

        if (A >= B && A >= C) {
            maior = A;
            menor = B < C ? B : C;
            meio = B > C ? B : C;
        } else if (B >= A && B >= C) {
            maior = B;
            menor = A < C ? A : C;
            meio = A > C ? A : C;
        } else {
            maior = C;
            menor = A < B ? A : B;
            meio = A > B ? A : B;
        }

        if (maior * maior == menor * menor + meio * meio) {
            printf("Triângulo Retângulo (ângulo de 90º).\n");
        } else if (maior * maior > menor * menor + meio * meio) {
            printf("Triângulo Obtusângulo (ângulo maior que 90º).\n");
        } else {
            printf("Triângulo Acutângulo (ângulo menor que 90º).\n");
        }
    } else {
        printf("Os valores não formam um triângulo.\n");
    }

}
